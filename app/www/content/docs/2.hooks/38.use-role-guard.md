
## Usage

```tsx [example.ts]
import { useRoleGuard } from "./{hooks file}";

const { hasAccess, user, isLoading } = useRoleGuard(["admin", "editor"], {
  redirectTo: "/login",
});
if (isLoading) return <Loader />;
if (!hasAccess) return null; // Will redirect automatically
return <AdminPanel user={user} />;
```
