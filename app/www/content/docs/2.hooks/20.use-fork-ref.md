## Installation

::code-group{sync="pm"}

```bash [npm]
npx sse-hooks add use-fork-ref
```

```bash [yarn]
yarn dlx sse-hooks add use-fork-ref
```

```bash [pnpm]
pnpm dlx sse-hooks add use-fork-ref
```

```bash [deno]
deno run -A npm:sse-hooks add use-fork-ref
```

```bash [bun]
bunx sse-hooks add use-fork-ref
```

::

## Usage

```tsx [example.ts]
import { useForkRef } from "./{hooks file}";

const rootRef = React.useRef<Instance>(null);
const refFork = useForkRef(rootRef, props.ref);

return <Root {...props} ref={refFork} />;
```
